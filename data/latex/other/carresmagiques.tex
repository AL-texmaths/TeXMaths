\def\carremagiqueB#1#2#3{%
\def\Ba{#1}
\def\Bc{#2}
\def\Ac{#3}

\def\Bb{\number\numexpr(\Ba+\Bc)/2\relax}
\def\squaresum{\number\numexpr\Ba+\Bb+\Bc\relax}
\def\Cc{\number\numexpr\Ba+\Bb-\Ac\relax}
\def\Aa{\number\numexpr\Ac+\Bc-\Bb\relax}
\def\Ab{\number\numexpr\squaresum-\Aa-\Ac\relax}
\def\Cb{\number\numexpr\squaresum-\Ab-\Bb\relax}
\def\Ca{\number\numexpr\squaresum-\Aa-\Ba\relax}

\begin{tblr}{colspec={*{3}{Q[c,m,wd=10pt]}}, hlines, vlines,rows={ht=16pt}}
    \@hideorshow\Aa&\@hideorshow\Ab&\bfseries\Ac\\
    \bfseries\Ba&\@hideorshow\Bb&\bfseries\Bc\\
    \@hideorshow\Ca&\@hideorshow\Cb&\@hideorshow\Cc
\end{tblr}}

\def\@show#1{#1}

\def\carremagiqueA#1#2#3{%
    \def\aaa{#1}
    \def\bbb{#2}
    \def\ccc{#3}
    \def\ddd{(\aaa+\bbb)/2}
    \def\eee{(\aaa+\ccc)/2}
    \def\fff{(\bbb+\ccc)/2}
    \def\ggg{(\bbb-\ccc)/2}
    \def\hhh{(\aaa-\ccc)/2}

    \begin{tblr}{colspec={*{3}{Q[c,m,wd=15pt]}}, hlines, vlines,rows={ht=21pt}}
        \@hideorshow{\number\numexpr\hhh+\bbb\relax}&\bfseries\number\numexpr\ccc\relax&\@hideorshow{\number\numexpr\ggg+\aaa\relax}\\
        \bfseries\number\numexpr\aaa\relax&\@hideorshow{\number\numexpr\ddd\relax}&\bfseries\number\numexpr\bbb\relax\\
        \@hideorshow{\number\numexpr\fff\relax}&\@hideorshow{\number\numexpr\aaa+\bbb-\ccc\relax}&\@hideorshow{\number\numexpr\eee\relax}\\
    \end{tblr}}
\makeatother
