\def\initlosanges{\coordinate(O)at(0,0);
\coordinate(A)at(150:0.866);

\makeatletter
\newcount\DNB@losanges
\def\losange(#1,#2){\advance\DNB@losanges1\relax
    \coordinate(O\number\DNB@losanges)at(#1,#2);
    \path(O\number\DNB@losanges)+(A)coordinate(A\number\DNB@losanges);
    \path(O\number\DNB@losanges)+(B)coordinate(B\number\DNB@losanges);
    \path(O\number\DNB@losanges)+(C)coordinate(C\number\DNB@losanges);
    \path(O\number\DNB@losanges)+(D)coordinate(D\number\DNB@losanges);
    \draw(A\number\DNB@losanges)--(B\number\DNB@losanges)--(C\number\DNB@losanges)--(D\number\DNB@losanges)--cycle;
}
