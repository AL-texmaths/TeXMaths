    \part{Médiatrice d'un segment}

        \subpart{Définition}
        \definition{mediatrice}
        \begin{exemple}
            \begin{center}
                \begin{minipage}[c]{.35\linewidth}
                    \exported[scale=1,lmargin=30,vmargins=30,active=1]%
                    <<
                        \begin{tikzpicture}[rotate=30]
                            \coordinate(O)at(0,0);
                            \coordinate[label=below left:A](A)at(-2,0);
                            \coordinate[label=right:B](B)at(2,0);
                            \coordinate[label=below left:\color{red}(d)]%
                            (C)at(0,3);
                            \coordinate(D)at(0,-2);
                            \tkzMarkRightAngle(B,O,C)
                            \path(A)--(O)node[pos=0.5]{||};
                            \path(O)--(B)node[pos=0.5]{||};
                            \draw(A)--(B);
                            \draw[red](C)--(D);
                            \node at(0,3.5){};
                        \end{tikzpicture}
                    >>
                \end{minipage}
                \begin{minipage}[c]{.55\linewidth}
                    La droite \textsf{(d)} est la médiatrice du segment
                    \textsf{[AB]}.
                \end{minipage}
            \end{center}
        \end{exemple}

        \subpart{Construction}
        \propriete{mediatrice_equidistance}
        \methode{mediatrice_construction_compas}
        \begin{exemple}
            \begin{center}
                \begin{tikzpicture}[rotate=30]
                    \coordinate(O)at(0,0);
                    \coordinate[label=below left:A](A)at(-2,0);
                    \coordinate[label=right:B](B)at(2,0);
                    \draw(A)--(B);
                    \tkzShowLine[red,mediator,color=gray,length=1](A,B)
                    \tkzGetPoints{i}{j}
                    \tkzDrawLines[red,add=-0.1 and -0.1](i,j)
                \end{tikzpicture}
            \end{center}
        \end{exemple}

        \part{Axes de symétries d'une figure}
            \definition{axe_de_symetrie}
            \begin{exemple}
                Les figures usuelles suivantes possèdent des axes de symétrie.
                \begin{center}
                    \exported[scale=1,lmargin=30,vmargins=30,active=1]%
                    <<
                        \begin{tblr}{
                            colspec={Q[c,m,wd=100pt]Q[c,m,wd=100pt]Q[c,m,wd=100pt]},
                            hline{4} = {2pt},
                            row{1} = {abovesep=10pt,belowsep=15pt},
                            row{2-3} = {abovesep=5pt,belowsep=5pt},
                            row{4} = {abovesep=15pt,belowsep=15pt},
                            row{5-6} = {abovesep=5pt,belowsep=5pt},
                            hlines,vlines,
                            row{odd} = {bg=Rainbow1-rouge!40!white},
                            row{even} = {bg=Rainbow1-rouge!70!white}%
                        }
                            \begin{tikzpicture}[baseline=(O.base)]
                                \coordinate(O)at(0,0);
                                \coordinate(A)at(0:1);
                                \coordinate(B0)at(140:1.2);
                                \coordinate(B1)at(-140:1.2);
                                \draw(A)--(B0)--(B1)--cycle;
                                \toComplete{%
                                    \draw[red](180:1.2)--(0:1.2);    
                                }
                            \end{tikzpicture}
                            &
                            \begin{tikzpicture}[baseline=(O.base)]
                                \coordinate(O)at(0,0);
                                \coordinate(A0)at(0:1);
                                \coordinate(B0)at(120:1);
                                \coordinate(B1)at(-120:1);
                                \draw(A0)--(B0)--(B1)--cycle;
                                \toComplete{%
                                    \path(O)--+(0:-0.7)coordinate(C0);
                                    \path(A0)--+(0:0.2)coordinate(C1);
                                    \draw[red](C0)--(C1);
                                    \path(O)--+(120:-0.7)coordinate(D0);
                                    \path(B0)--+(120:0.2)coordinate(D1);
                                    \draw[red](D0)--(D1);
                                    \path(O)--+(-120:-0.7)coordinate(E0);
                                    \path(B1)--+(-120:0.2)coordinate(E1);
                                    \draw[red](E0)--(E1);
                                }
                            \end{tikzpicture}
                            &
                            \begin{tikzpicture}[baseline=(O.base)]
                                \coordinate(O)at(0,0);
                                \coordinate(A0)at(45:1);
                                \coordinate(A1)at(135:1);
                                \coordinate(A2)at(-135:1);
                                \coordinate(A3)at(-45:1);
                                \draw(A0)--(A1)--(A2)--(A3)--cycle;
                                \toComplete{%
                                    \path(A0)--+(45:0.3)coordinate(B0);
                                    \path(A2)--+(-135:0.3)coordinate(B2);
                                    \draw[red](B0)--(B2);
                                    \path(A1)--+(135:0.3)coordinate(B1);
                                    \path(A3)--+(-45:0.3)coordinate(B3);
                                    \draw[red](B1)--(B3);
                                    \draw[red](-1,0)--(1,0);
                                    \draw[red](0,-1)--(0,1);
                                }
                            \end{tikzpicture}
                            \\
                            \toComplete{triangle isocèle}
                            &\toComplete{triangle équilatéral}
                            &\toComplete{carré}\\
                            \toComplete{1 axe de symétrie}
                            &\toComplete{3 axes de symétrie}
                            &\toComplete{4 axe de symétrie}\\
                            \begin{tikzpicture}[baseline=(O.base)]
                                \coordinate(O)at(0,0);
                                \coordinate(A0)at(30:1);
                                \coordinate(A1)at(150:1);
                                \coordinate(A2)at(-150:1);
                                \coordinate(A3)at(-30:1);
                                \draw(A0)--(A1)--(A2)--(A3)--cycle;
                                \toComplete{%
                                    \draw[red](90:0.9)--(-90:0.9);
                                    \draw[red](0:1.2)--(180:1.2);
                                }
                            \end{tikzpicture}
                            &
                            \begin{tikzpicture}[baseline=(O.base)]
                                \coordinate(O)at(0,0);
                                \coordinate(A0)at(0:1);
                                \coordinate(A1)at(90:0.7);
                                \coordinate(A2)at(180:1);
                                \coordinate(A3)at(-90:0.7);
                                \draw(A0)--(A1)--(A2)--(A3)--cycle;
                                \toComplete{%
                                    \draw[red](90:0.9)--(-90:0.9);
                                    \draw[red](0:1.2)--(180:1.2);
                                }
                            \end{tikzpicture}
                            &
                            \begin{tikzpicture}[baseline=(O.base)]
                                \coordinate(O)at(0,0);
                                \draw(O)circle(0.8);
                                \toComplete{%
                                    \foreach\circleAxeAngle in{0,30,...,330}{%
                                        \draw[red](\circleAxeAngle:1.2)--(\circleAxeAngle+180:1.2);%
                                    }
                                }
                            \end{tikzpicture}
                            \\
                            \toComplete{rectangle}
                            &\toComplete{losange}
                            &\toComplete{cercle}\\
                            \toComplete{2 axes de symétrie}
                            &\toComplete{2 axes de symétrie}
                            &\toComplete{une infinité d'axes de symétrie}\\
                        \end{tblr}
                    >>
                \end{center}
            \end{exemple}
