        \subpart{Périmètre d'un cercle}
            \propriete{perimetre_cercle}
            \begin{remarque}
                Le nombre $\pi$ (prononcé \guillemets{pi}) est défini par la
                formule ci-dessus. Il vaut environ 3,14.
            \end{remarque}
            \begin{exemple}
                \exported[scale=1,lmargin=30,vmargins=30]%
                    <<
                        \par\noindent
                        \begin{minipage}[c]{.45\linewidth}
                            \begin{tikzpicture}
                                \coordinate[label=below left:O](O)at(0,0);
                                \draw(O)circle(2.6);
                                \coordinate[label=above right:A](A)at(30:2.6);
                                \node at(A){$\times$};
                                \node at(O){$\times$};
                                \path(O)--(A)node[pos=0.5,sloped,above]{2,6 cm};
                                \draw(O)--(A);
                                \node at(0,3){\phantom{a}};
                            \end{tikzpicture}
                        \end{minipage}\hspace{.08\linewidth}
                        \begin{minipage}[c]{.45\linewidth}
                            \toComplete{Le rayon de ce cercle vaut 2,6 cm. Donc son diamètre vaut 5,2~cm.
                            \vspace{20pt}
                            \begingroup\renewcommand{\arraystretch}{1.3}$\begin{array}{ccccc}
                                \textsf{périmètre cercle}&=&\textsf{diamètre}&\times&\pi\\
                                &=&5,2&\times&\pi\\
                                &\approx&16,34&
                            \end{array}$\endgroup\vspace{10pt}
                            Le périmètre du cercle vaut environ 16,34~cm.}
                        \end{minipage}
                    >>
            \end{exemple}
