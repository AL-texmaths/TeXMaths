% Hexagone régulier, sommet en haut
% #1 (optionnel) : rayon en cm (défaut : 1)
% #2 : coordonnées du centre (ex. "0,0")
\newcommand{\hexagone}[2][1]{%
  \begin{scope}[shift={(#2)}]
    \draw (90:#1) -- (30:#1) -- (-30:#1) -- (-90:#1) -- (-150:#1) -- (150:#1) -- cycle;
  \end{scope}%
}
\def\identity#1{#1}
