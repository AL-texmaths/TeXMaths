\def\fraction#1/#2,scale=#3,color=#4,rotate=#5;{%
    \begin{tikzpicture}[scale=#3,rotate=#5]
        \coordinate(O)at(0,0);
        \foreach\i in{0,...,#2}{
            \coordinate(A\i)at(90+\i*360/#2:1);
        }
        \draw(O)circle(1);
        \edef\jmax{\number\numexpr#1-1\relax}
        \foreach\j in{0,...,\jmax}{%
            \filldraw[fill=#4](O)--(A\j)arc[%
                start angle=90+\j*360/#2,
                end angle=90+(\j+1)*360/#2,
                radius=1]--cycle;
            }
        \foreach\i in{1,...,#2}{%
            \draw(O)--(A\i);
        }
    \end{tikzpicture}%
    }
