    \coordinate(d11)at(5,0);
    \coordinate[label=right:$\sf(d_2)$](d20)at(0,5);

    \tkzDefPointsBy[reflection = over d20--d21](D,E,F){G,H,I}

    \draw(O)--(A)--(B)--(C)--(O);

    \draw[red](d20)--(d21);
}
