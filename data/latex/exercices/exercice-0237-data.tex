Soit $f$ une fonction.\par\smallskip\noindent On donne :\qquad
$f(-2)=0$\qquad
$f:-1\mapsto2$\qquad
$f(0)=4$\qquad
$f:1\mapsto6$\qquad
$f(2)=8$\qquad
\begin{enumerate}
    \item 0 est un antécedent de 4 par $f$ ;\par\noindent\hbox to.75\linewidth{$\square$\quad VRAI\hfill$\square$\quad FAUX\hfill$\square$\quad On ne peut pas savoir}
    \item 2 a pour image -1 par $f$ ;\par\noindent\hbox to.75\linewidth{$\square$\quad VRAI\hfill$\square$\quad FAUX\hfill$\square$\quad On ne peut pas savoir}
    \item 1 a pour antécédent 6 par $f$ ;\par\noindent\hbox to.75\linewidth{$\square$\quad VRAI\hfill$\square$\quad FAUX\hfill$\square$\quad On ne peut pas savoir}
    \item 0 est l'image de -2 par $f$ ;\par\noindent\hbox to.75\linewidth{$\square$\quad VRAI\hfill$\square$\quad FAUX\hfill$\square$\quad On ne peut pas savoir}
    \item 2 est le seul antécédent de 8 par $f$ ;\par\noindent\hbox to.75\linewidth{$\square$\quad VRAI\hfill$\square$\quad FAUX\hfill$\square$\quad On ne peut pas savoir}
\end{enumerate}
