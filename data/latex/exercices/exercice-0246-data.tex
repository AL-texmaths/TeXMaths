Soit $g$ une fonction. On donne :\qquad
$g(-2)=3$\qquad
$g:-1\mapsto5$\qquad
$g(0)=1$\qquad
$g:1\mapsto7$\qquad
$g(2)=7$\qquad
\begin{enumerate}
    \item 0 est un antécedent de 1 par $g$ ;\par\noindent\hbox to.75\linewidth{$\square$\quad VRAI\hfill$\square$\quad FAUX\hfill$\square$\quad On ne peut pas savoir}
    \item 5 a pour image -1 par $g$ ;\par\noindent\hbox to.75\linewidth{$\square$\quad VRAI\hfill$\square$\quad FAUX\hfill$\square$\quad On ne peut pas savoir}
    \item 1 a pour antécédent 0 par $g$ ;\par\noindent\hbox to.75\linewidth{$\square$\quad VRAI\hfill$\square$\quad FAUX\hfill$\square$\quad On ne peut pas savoir}
    \item 3 est l'image de -2 par $g$ ;\par\noindent\hbox to.75\linewidth{$\square$\quad VRAI\hfill$\square$\quad FAUX\hfill$\square$\quad On ne peut pas savoir}
    \item 1 est le seul antécédent de 7 par $g$ ;\par\noindent\hbox to.75\linewidth{$\square$\quad VRAI\hfill$\square$\quad FAUX\hfill$\square$\quad On ne peut pas savoir}
\end{enumerate}
