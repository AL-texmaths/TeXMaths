\def\figureA{%
    \begin{tikzpicture}[baseline=(O.base),scale=\scale]
        \coordinate(O)at(0,0);
        \coordinate(A0)at(-1,-1);
        \path(A0)--+(-1,0)coordinate(A1);
        \path(A1)--+(0,3)coordinate(A2);
        \path(A2)--+(-1,0)coordinate(A3);
        \path(A3)--+(0,-4)coordinate(A4);
        \path(A4)--+(2,0)coordinate(A5);
        \coordinate(A)at(4,0);
        \tkzDefPointsBy[translation=from O to A](A0,A1,A2,A3,A4,A5){B0,B1,B2,B3,B4,B5}
        \filldraw[fill=Rainbow1-vert](A0)\foreach\num in{1,2,...,5}%
            {--(A\num)}--cycle;
        \filldraw[fill=Rainbow1-bleu](B0)\foreach\num in{1,2,...,5}%
            {--(B\num)}--cycle;
        \draw[gray](-4,-3)grid(5,4);
        \draw(A0)\foreach\num in{1,2,...,5}%
            {--(A\num)}--cycle;
        \draw(B0)\foreach\num in{1,2,...,5}%
            {--(B\num)}--cycle;
        \draw[red](0,-3)--(0,4);
    \end{tikzpicture}
}%
