    \qcmquestion{Question 4}
    On considère les nombres suivants et on s'intéresse à leur divisibilité par 9.

    \[N = \np{2025} \quad \text{et} \quad P = \np{2026}.\]

    \begin{qcm}
        \small $N$ et $P$ sont tous les deux divisibles par 9 & $N$ est divisible
        par 9 mais $P$ ne l'est pas & $P$ est divisible par 9 mais $N$ ne l'est pas &
        Aucun des deux n'est divisible par 9
    \end{qcm}
