\begin{document}
    \begin{flushleft}
        Transformer les expressions littérales suivantes en supprimant (si possible) les $\times$.
    \end{flushleft}
    \hrule\vskip10pt\par
        \textbf{Règle : }On peut supprimer le symbole $\times$ devant une
        lettre.
    \vskip10pt\hrule
    \begin{enumerate}
        \item $4\times a$\vskip30pt
        \item $b\times 16$\vskip30pt
        \item $x\times y$\vskip30pt
    \end{enumerate}
    \hrule\vskip10pt\par
        \textbf{Règle : }On peut supprimer le symbole $\times$ devant des
        parenthèses.
        \vskip10pt\hrule
    \begin{enumerate}
        \item $2\times(e+2)$\vskip30pt
        \item $3\times(1-5\times x)$\vskip30pt
    \end{enumerate}
    \hrule\vskip10pt\par
        \centering Cas particuliers :
        \vskip10pt\hrule
    \begin{enumerate}
        \item $1\times a$\vskip30pt
        \item $r\times r$\vskip30pt
        \item $c\times c\times c$
    \end{enumerate}
