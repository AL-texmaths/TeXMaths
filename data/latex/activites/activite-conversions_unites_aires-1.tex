    \begin{enumerate}
        \item On souhaite convertir 34,5 m$^2$ en cm$^2$.
        \begin{enumerate}
            \item Je place le chiffre des unités dans la {\bfseries colonne de
            droite} des m$^2$.
            \item Je complète le nombre sans ajouter la virgule.
            \item Je m'assure qu'il y a bien un chiffre dans la colonne de
            droite des cm$^2$.
            \item Je complète en utilisant des zéros.
            \item La colonne de droite de la nouvelle unité contient le chiffre des
            unités.\vskip20pt
            Donc 34,5 m$^2=$\doted{30pt}$^2$
        \end{enumerate}
        \item Convertir.

            \item $12,5\,\text{dam}^2=\doted{50pt}\,\text{km}^2$
            \item $12\,472\,\text{dm}^2=\doted{50pt}\,\text{m}^2$
        \end{enumerate}
    \end{enumerate}
