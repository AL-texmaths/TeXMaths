\textbf{Définitions:}
\begin{enumerate}[label=$ $]
    \item On dit qu'un angle est \textbf{aigu} s'il est plus petit qu'un angle droit.
    \item On dit qu'un angle est \textbf{plat} s'il ses côtés forment une ligne droite.
    \item On dit qu'un angle est \textbf{obtus} s'il est plus grand qu'un angle droit et plus petit qu'un angle plat.
\end{enumerate}
\rule{\textwidth}{0.4pt}
