{\bfseries\centering Première partie\par}\noindent
\begin{enumerate}
    \item Dans le cahier d'exercice, construire un triangle \mathspoint{ABC}
    rectangle en \mathspoint{B} tel que~:
    \begin{itemize}
        \item $\widehat{\mathspoint{BAC}}=40^\circ$
        \item $\mathspoint{AC} = 1 \text{ dm}$
    \end{itemize}
    \item Coder la figure.
    \item Sur la figure, noter:
    \begin{itemize}
        \item la mesure du côté adjacent à l'angle;
        \item la mesure du côté opposé à l'angle.
    \end{itemize}
\end{enumerate}
{\bfseries\centering Deuxième partie\par}\noindent
\begin{enumerate}
    \item Sur la calculatrice, trouver les touches \guillemets{cos}
    et \guillemets{sin}.
    \item Calculer (arrondir au dixième)~:
    \begin{itemize}
        \item $\text{cos}(40)\approx$
        \item $\text{sin}(40)\approx$
    \end{itemize}
    \item Comparer vos résultats aux mesures à la dernière question de la
    première partie. Que remarquez-vous~?
    \answerspace{5cm}
\end{enumerate}
