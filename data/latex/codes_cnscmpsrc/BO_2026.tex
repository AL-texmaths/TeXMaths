\def\write@key@value#1#2{%
    {\textcolor{\getColor\levelfor!60!black}{\ttfamily #1}\hskip10pt}#2\vskip8pt
}

\newcommand{\getColor}[1]{%
    \ifcase#1\relax
        black\or
        Rainbow1-bleu\or
        Rainbow1-vert\or
        Rainbow1-orange%
    \else
        black%
    \fi
}

\def\showtitle#1{%
    \begin{tcolorbox}[
            colback=\getColor\levelfor!25!white,
            colframe=\getColor\levelfor!75!black
            ]
        \texttt{#1}
    \end{tcolorbox}
}

\def\showitems{%
    \foreach\themefor in{1,...,\cycle@total[themes]}
    {%
        \newpage
        \def\themekey{\nbtoAlph\themefor}
        % register theme in ToC (only themes will be added)
        \phantomsection
        \addcontentsline{toc}{theme}{Thème \themekey\ : \cycle@theme[\themekey]}
        {\par\centering\LARGE\bfseries Thème \themekey\ : \cycle@theme[\themekey]\centering
            \par}\vskip8pt
        \foreach\levelfor in{1,...,\cycle@total[\nbtoAlph\themefor]}
        {%
            \def\levelkey{\nbtoAlph\themefor\levelfor}
            \phantomsection
            \addcontentsline{toc}{level}{\protect\hspace*{2em}\small\cycle@level[\levelkey]}
            \tcbox[colback=\getColor\levelfor!25!white]{\ttfamily\cycle@level[\levelkey]}
            \foreach\subthemefor in{1,...,\cycle@total[\nbtoAlph\themefor\levelfor]}
            {%
                \def\subthemekey{\nbtoAlph\themefor\levelfor\nbtoalph\subthemefor}%
                \noindent\textbf{Sous-thème \subthemefor~: \cycle@subtheme[\subthemekey]}\par\medskip\noindent
            \edef\cycle@total@aut{\cycle@total[\nbtoAlph\themefor\levelfor\nbtoalph\subthemefor AUT]}%
            \ifnum\cycle@total@aut>0\relax
                \showtitle{Automatismes}
                \foreach\autfor in{1,...,\cycle@total@aut}{%
                    \def\autkey{\nbtoAlph\themefor\levelfor\nbtoalph\subthemefor\nbtoalph\autfor}%
                    \write@key@value{\autkey}{\cycle@aut[\autkey]}
                }%
            \fi
            \edef\cycle@total@obj{\cycle@total[\nbtoAlph\themefor\levelfor\nbtoalph\subthemefor OBJ]}%
            \ifnum\cycle@total@obj>0\relax
                \showtitle{Objectifs d'aprentissages}
                \foreach\objfor in{1,...,\cycle@total@obj}{%
                    \def\objkey{\nbtoAlph\themefor\levelfor\nbtoalph\subthemefor\nbtoalph\objfor}%
                    \write@key@value{\objkey}{\cycle@obj[\objkey]}
                }%
            \fi
            \edef\cycle@total@pro{\cycle@total[\nbtoAlph\themefor\levelfor\nbtoalph\subthemefor PRO]}%
            \ifnum\cycle@total@pro>0\relax
                \showtitle{Prolongements}
                \foreach\profor in{1,...,\cycle@total@pro}{%
                    \def\prokey{\nbtoAlph\themefor\levelfor\nbtoalph\subthemefor\nbtoalph\profor}%
                    \write@key@value{\prokey}{\cycle@pro[\prokey]}
                }%
            \fi
            }%
        }%
    }%
    \vskip0pt plus 3fill\begingroup\large\tt
    D'après le BO n$^\circ$ \boNum\endgroup
    \vskip0pt plus 1fill
}
