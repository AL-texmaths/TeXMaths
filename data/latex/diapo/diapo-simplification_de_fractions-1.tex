\def\fracframe#1{%
    \edef\numerateur{\number\numexpr#1*2\relax}
    \edef\denominateur{\number\numexpr#1*3\relax}
    \begin{frame}
        \fraction2/3,scale=1;
        \begin{tblr}{ccc}
            &\color{vscode-green}\Large$\times#1$&\\
            \SetCell[c=3]{l,m}\newuparrow[line width=1.5pt, vscode-green]{1}&&\\
            $\dfrac23$&$=$&$\dfrac{\numerateur}{\denominateur}$\\
            \SetCell[c=3]{l,m}\newdownarrow[line width=1.5pt, vscode-green]{1}&&\\
            &\color{vscode-green}\Large$\times#1$&\\
        \end{tblr}
        \fraction\numerateur/\denominateur,scale=1;
    \end{frame}
}
