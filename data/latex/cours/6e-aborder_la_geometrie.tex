    \bigpart{Alignement et appartenance}
        \definition{appartient}
        {\bfseries Exemple :}\par\kern5pt\global\leftskip2\parindent\noindent
        \begin{tikzpicture}[baseline=(A.base)]
            \coordinate[label=above left:A](A)at(0,0);
            \coordinate[label=above left:B](B)at(2,1);
            \coordinate[label=above right:C](C)at(3,1.5);
            \coordinate[label=above right:K](K)at(2.5,0);
            \draw(A)--(C);
            \draw node at(A){+};
            \draw node at(B){+};
            \draw node at(C){+};
            \draw node at(K){+};
        \end{tikzpicture}\vskip10pt
        \noindent
        Le point B appartient au segment d'extrémités les points A et C.
        $$\text{B}\in[\text{AC}]$$
        Le point K n'appartient pas à la droite passant par les points A et B.
        $$\text{K}\notin(\text{AB})$$
        \global\leftskip\parindent
        \definition{alignes}
        {\bfseries Exemple :}\par\kern5pt\global\leftskip2\parindent\noindent
        \begin{tikzpicture}[baseline=(A.base)]
            \coordinate[label=above left:A](A)at(0,0);
            \coordinate[label=above left:B](B)at(2,1);
            \coordinate[label=above right:C](C)at(3,1.5);
            \coordinate[label=above right:K](K)at(2.5,0);
            \draw(A)--(C);
            \draw node at(A){+};
            \draw node at(B){+};
            \draw node at(C){+};
            \draw node at(K){+};
        \end{tikzpicture}\vskip10pt
        \noindent Les points A, B et C sont {\bfseries alignés} car ils se trouvent
        sur la même droite.\vskip5pt\noindent Les points A, B, C et K ne sont pas
        alignés.

    \bigpart{Programme de tracé}
        Avant de réaliser un programme de tracé, on fait une figure à main levée.
        \begin{mdframed}
            \begin{enumerate}
            \item Placer trois points A, B et C non alignés. 
            \item Placer un point D sur le segment [AB]. 
            \item Placer le point K, milieu du segment [AC]. 
            \item Enfin, tracer les droites (DC) et (KB). 
            \item Ces deux droites se coupent en M. Placer le point M.
            \end{enumerate}
        \end{mdframed}
