    \bigpart{Préambule}
    \proposition{a_times_x_equal_b}
    \noindent{\bfseries Exemple :}\vskip10pt
    $8\times?=72$\par\noindent La réponse est le nombre $\dfrac{72}8$.
    Comme $72\div8=9$, il s'agit du nombre 9 :$$8\times9=72$$\vskip10pt
    $6\times?=9$\par\noindent La réponse est le nombre $\dfrac{9}6$.
    Comme $9\div6=1,5$, il s'agit du nombre 1,5 :$$6\times1,5=9$$\vskip10pt
    $3\times?=1$\par\noindent La réponse est le nombre $\dfrac{1}3$.
    $$3\times\dfrac{1}3=1$$
