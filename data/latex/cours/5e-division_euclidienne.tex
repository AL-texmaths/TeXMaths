    \bigpart{Vocabulaire}
    \methode{tester_diviseur}
    \definition{multiple}
    \par\noindent{\bfseries Exemple :}\vskip10pt\noindent
        On sait que $27=3\times9$ donc 27 est dans la table de 9.\par\noindent
        On dit que 27 est un multiple de 9.
    \definition{diviseur}
    \par\noindent{\bfseries Exemple :}\vskip10pt\noindent
        On sait que $40=8\times5$ donc 40 est dans la table de 5.\par\noindent
        On dit que 5 est un diviseur de 40.
    \par\noindent{\bfseries Exemple :}\vskip10pt\noindent
    On sait que $$24=6\times4$$.\par\noindent
    Donc 24 est un multiple de 6 et de 4.\par\noindent
    Donc 6 et 4 sont des diviseurs de 24.
