    \subpart{Calculer une longueur}\bigskip
    {\bfseries Exemple :}\par\medskip
    \begin{minipage}{.35\linewidth}
        \begin{tikzpicture}
            \coordinate[label=below left:A](A)at(0,0);
            \coordinate[label=below right:B](B)at(3,0);
            \coordinate[label=C](C)at(3,2);
            \tkzMarkRightAngle[size=0.2](C,B,A)
            \tkzLabelAngle[pos=1.25](B,A,C){\bfseries 40$^\circ$}
            \tkzFillAngle[size=0.8,fill=black!25!white](B,A,C)
            \draw(A)--(B)--(C)--cycle;
            \path(A)--(C)node[midway,above left]{\bfseries 8};
            \path(B)--(C)node[midway,right]{\bfseries ?};
        \end{tikzpicture}
    \end{minipage}%
    \begin{minipage}{.5\linewidth}
        \begin{align*}
            \text{sin}(40^\circ)&\;=\;\dfrac{\text{BC}}8\\
            \text{BC}&\;=\;8\times\text{sin}(40^\circ)\\
            \text{BC}&\;\approx\;5,14
        \end{align*}
    \end{minipage}\par\bigskip
