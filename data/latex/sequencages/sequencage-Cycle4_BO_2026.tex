\def\BoItems#1#2#3{%
    {\itshape Automatismes}\vskip8pt
    \@for\ii:=#1\do{%
        \par\noindent$-$\hskip2ex\useaut[\ii]%
        \immediate\write\seqcanal{AUT\ii:\@level@name:\sequence@key:}%
    }\vskip5pt\noindent\nopagebreak
    % -----------------------------
    {\itshape Objectifs d'apprentissage}\vskip8pt
    \@for\ii:=#2\do{%
        \par\noindent$-$\hskip2ex\useobj[\ii]%
        \immediate\write\seqcanal{OBJ\ii:\@level@name:\sequence@key:}%
    }\vskip5pt\noindent\nopagebreak
    % -----------------------------
    {\itshape Prolongements}\vskip8pt
    \@for\ii:=#3\do{%
        \par\noindent$-$\hskip2ex\usepro[\ii]%
        \immediate\write\seqcanal{PRO\ii:\@level@name:\sequence@key:}%
    }\vskip5pt\noindent\nopagebreak
    \par\kern20pt\noindent
}
